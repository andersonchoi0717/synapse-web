import openai
from flask import Flask, request, jsonify
import random

app = Flask(__name__)

# Predefined emotion-to-color mapping
emotion_colors = {
"happy": "#FFD700", # Gold
"sad": "#4682B4", # Steel Blue
"angry": "#FF4500", # Orange Red
"calm": "#87CEEB", # Sky Blue
"excited": "#FF69B4", # Hot Pink
"nostalgic": "#8B4513", # Saddle Brown
"anxious": "#8A2BE2", # Blue Violet
}

# Predefined emotion-to-music mapping
emotion_music = {
"happy": "Upbeat piano with cheerful violins",
"sad": "Slow cello with soft rain sounds",
"angry": "Fast drum beats with intense electric guitar",
"calm": "Gentle harp with ocean waves",
"excited": "Electronic dance beats with bright synths",
"nostalgic": "Soft jazz with vinyl crackling",
"anxious": "Repetitive clock ticking with eerie synths",
}

# Function to generate AI poetic response
def generate_poetic_response(emotion):
prompt = f"Create a short poetic reflection for someone feeling {emotion}."
response = openai.ChatCompletion.create(
model="gpt-4",
messages=[{"role": "system", "content": "You are a poetic AI that responds with artistic and emotional language."},
{"role": "user", "content": prompt}]
)
return response['choices'][0]['message']['content']

@app.route('/emotion', methods=['POST'])
def process_emotion():
data = request.json
emotion = data.get("emotion", "calm").lower()
color = emotion_colors.get(emotion, "#FFFFFF") # Default to white
music = emotion_music.get(emotion, "Soft ambient tones")
poetry = generate_poetic_response(emotion)
return jsonify({
"emotion": emotion,
"color": color,
"music": music,
"poetic_response": poetry
})

if __name__ == '__main__':
app.run(debug=True)
